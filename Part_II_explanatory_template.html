# Set the overall aesthetic.
sns.set(style="whitegrid")

# Create a figure and a set of subplots
fig, ax = plt.subplots(figsize=(10, 6))

# Create a boxplot
sns.boxplot(data=df, x='user_type', y='duration_sec', ax=ax)

# Add median values on top of the boxplot
medians = df.groupby(['user_type'])['duration_sec'].median().values
nobs = df['user_type'].value_counts().values
nobs = [str(x) for x in nobs.tolist()]
nobs = ["n: " + i for i in nobs]

pos = range(len(nobs))
for tick, label in zip(pos, ax.get_xticklabels()):
    ax.text(pos[tick], medians[tick] + 0.03, nobs[tick],
            horizontalalignment='center', size='x-small', color='w', weight='semibold')

# Set the title and labels
ax.set_title('Boxplot of duration_sec for each user_type', fontsize=15)
ax.set_xlabel('User Type', fontsize=12)
ax.set_ylabel('Duration (sec)', fontsize=12)

# Limit y-axis for better visualization
ax.set_ylim([0, 5000])

# Remove the top and right spines from plot
sns.despine()

# Show the plot
plt.show()

In this graph you can see very clearly that subscribers have shorter average times than consumers.
